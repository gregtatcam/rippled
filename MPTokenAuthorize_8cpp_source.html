29 MPTokenAuthorize::preflight(PreflightContext const& ctx)
30 {
31  if (!ctx.rules.enabled(featureMPTokensV1))
32  return temDISABLED;
33 
34  if (auto const ret = preflight1(ctx); !isTesSuccess(ret))
35  return ret;
36 
37  if (ctx.tx.getFlags() & tfMPTokenAuthorizeMask)
38  return temINVALID_FLAG;
39 
40  if (ctx.tx[sfAccount] == ctx.tx[~sfMPTokenHolder])
41  return temMALFORMED;
42 
43  return preflight2(ctx);
44 }

47 MPTokenAuthorize::preclaim(PreclaimContext const& ctx)
48 {
49  auto const accountID = ctx.tx[sfAccount];
50  auto const holderID = ctx.tx[~sfMPTokenHolder];
51 
52  if (holderID && !(ctx.view.exists(keylet::account(*holderID))))
53  return tecNO_DST;
54 
55  // if non-issuer account submits this tx, then they are trying either:
56  // 1. Unauthorize/delete MPToken
57  // 2. Use/create MPToken
58  //
59  // Note: `accountID` is holder's account
60  // `holderID` is NOT used
61  if (!holderID)
62  {
63  std::shared_ptr<SLE const> sleMpt = ctx.view.read(
64  keylet::mptoken(ctx.tx[sfMPTokenIssuanceID], accountID));
65 
66  // There is an edge case where holder deletes MPT after issuance has
67  // already been destroyed. So we must check for unauthorize before
68  // fetching the MPTIssuance object(since it doesn't exist)
69 
70  // if holder wants to delete/unauthorize a mpt
71  if (ctx.tx.getFlags() & tfMPTUnauthorize)
72  {
73  if (!sleMpt)
74  return tecOBJECT_NOT_FOUND;
75 
76  if ((*sleMpt)[sfMPTAmount] != 0)
77  return tecHAS_OBLIGATIONS;
78 
79  return tesSUCCESS;
80  }
81 
82  // Now test when the holder wants to hold/create/authorize a new MPT
83  auto const sleMptIssuance =
84  ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
85 
86  if (!sleMptIssuance)
87  return tecOBJECT_NOT_FOUND;
88 
89  if (accountID == (*sleMptIssuance)[sfIssuer])
90  return temMALFORMED;
91 
92  // if holder wants to use and create a mpt
93  if (sleMpt)
94  return tecMPTOKEN_EXISTS;
95 
96  return tesSUCCESS;
97  }
98 
99  auto const sleMptIssuance =
100  ctx.view.read(keylet::mptIssuance(ctx.tx[sfMPTokenIssuanceID]));
101  if (!sleMptIssuance)
102  return tecOBJECT_NOT_FOUND;
103 
104  std::uint32_t const mptIssuanceFlags = sleMptIssuance->getFieldU32(sfFlags);
105 
106  // If tx is submitted by issuer, they would either try to do the following
107  // for allowlisting:
108  // 1. authorize an account
109  // 2. unauthorize an account
110  //
111  // Note: `accountID` is issuer's account
112  // `holderID` is holder's account
113  if (accountID != (*sleMptIssuance)[sfIssuer])
114  return temMALFORMED;
115 
116  // If tx is submitted by issuer, it only applies for MPT with
117  // lsfMPTRequireAuth set
118  if (!(mptIssuanceFlags & lsfMPTRequireAuth))
119  return tecNO_AUTH;
120 
121  if (!ctx.view.exists(
122  keylet::mptoken(ctx.tx[sfMPTokenIssuanceID], *holderID)))
123  return tecOBJECT_NOT_FOUND;
124 
125  return tesSUCCESS;
126 }

129 MPTokenAuthorize::doApply()
130 {
131  auto const mptIssuanceID = ctx_.tx[sfMPTokenIssuanceID];
132  auto const sleAcct = view().peek(keylet::account(account_));
133  if (!sleAcct)
134  return tecINTERNAL;
135 
136  auto const holderID = ctx_.tx[~sfMPTokenHolder];
137 
138  // If the account that submitted the tx is a holder
139  // Note: `account_` is holder's account
140  // `holderID` is NOT used
141  if (!holderID)
142  {
143  // When a holder wants to unauthorize/delete a MPT, the ledger must
144  // - delete mptokenKey from owner directory
145  // - delete the MPToken
146  if (ctx_.tx.getFlags() & tfMPTUnauthorize)
147  {
148  auto const mptokenKey = keylet::mptoken(mptIssuanceID, account_);
149  auto const sleMpt = view().peek(mptokenKey);
150  if (!sleMpt)
151  return tecINTERNAL;
152 
153  if (!view().dirRemove(
154  keylet::ownerDir(account_),
155  (*sleMpt)[sfOwnerNode],
156  sleMpt->key(),
157  false))
158  return tecINTERNAL;
159 
160  adjustOwnerCount(view(), sleAcct, -1, j_);
161 
162  view().erase(sleMpt);
163  return tesSUCCESS;
164  }
165 
166  // A potential holder wants to authorize/hold a mpt, the ledger must:
167  // - add the new mptokenKey to the owner directory
168  // - create the MPToken object for the holder
169  std::uint32_t const uOwnerCount = sleAcct->getFieldU32(sfOwnerCount);
170  XRPAmount const reserveCreate(
171  (uOwnerCount < 2) ? XRPAmount(beast::zero)
172  : view().fees().accountReserve(uOwnerCount + 1));
173 
174  if (mPriorBalance < reserveCreate)
175  return tecINSUFFICIENT_RESERVE;
176 
177  auto const mptokenKey = keylet::mptoken(mptIssuanceID, account_);
178 
179  auto const ownerNode = view().dirInsert(
180  keylet::ownerDir(account_), mptokenKey, describeOwnerDir(account_));
181 
182  if (!ownerNode)
183  return tecDIR_FULL;
184 
185  auto mptoken = std::make_shared<SLE>(mptokenKey);
186  (*mptoken)[sfAccount] = account_;
187  (*mptoken)[sfMPTokenIssuanceID] = mptIssuanceID;
188  (*mptoken)[sfFlags] = 0;
189  (*mptoken)[sfOwnerNode] = *ownerNode;
190  view().insert(mptoken);
191 
192  // Update owner count.
193  adjustOwnerCount(view(), sleAcct, 1, j_);
194 
195  return tesSUCCESS;
196  }
197 
198  auto const sleMptIssuance = view().read(keylet::mptIssuance(mptIssuanceID));
199  if (!sleMptIssuance)
200  return tecINTERNAL;
201 
202  // If the account that submitted this tx is the issuer of the MPT
203  // Note: `account_` is issuer's account
204  // `holderID` is holder's account
205  if (account_ != (*sleMptIssuance)[sfIssuer])
206  return tecINTERNAL;
207 
208  auto const sleMpt = view().peek(keylet::mptoken(mptIssuanceID, *holderID));
209  if (!sleMpt)
210  return tecINTERNAL;
211 
212  std::uint32_t const flagsIn = sleMpt->getFieldU32(sfFlags);
213  std::uint32_t flagsOut = flagsIn;
214 
215  // Issuer wants to unauthorize the holder, unset lsfMPTAuthorized on
216  // their MPToken
217  if (ctx_.tx.getFlags() & tfMPTUnauthorize)
218  flagsOut &= ~lsfMPTAuthorized;
219  // Issuer wants to authorize a holder, set lsfMPTAuthorized on their
220  // MPToken
221  else
222  flagsOut |= lsfMPTAuthorized;
223 
224  if (flagsIn != flagsOut)
225  sleMpt->setFieldU32(sfFlags, flagsOut);
226 
227  view().update(sleMpt);
228  return tesSUCCESS;
229 }
