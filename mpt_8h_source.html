221  submit(A const& arg, Json::Value const& jv)
222  {
223  if (arg.err)
224  {
225  if (arg.flags && arg.flags > 0)
226  env_(jv, txflags(*arg.flags), ter(*arg.err));
227  else
228  env_(jv, ter(*arg.err));
229  }
230  else if (arg.flags && arg.flags > 0)
231  env_(jv, txflags(*arg.flags));
232  else
233  env_(jv);
234  auto const err = env_.ter();
235  if (close_)
236  env_.close();
237  if (arg.ownerCount)
238  env_.require(owners(issuer_, *arg.ownerCount));
239  if (arg.holderCount)
240  {
241  for (auto it : holders_)
242  env_.require(owners(*it.second, *arg.holderCount));
243  }
