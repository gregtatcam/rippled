216 inline MPTAmount
217 mulRatio(
218  MPTAmount const& amt,
219  std::uint32_t num,
220  std::uint32_t den,
221  bool roundUp)
222 {
223  using namespace boost::multiprecision;
224 
225  if (!den)
226  Throw<std::runtime_error>("division by zero");
227 
228  int128_t const amt128(amt.mpt());
229  auto const neg = amt.mpt() < 0;
230  auto const m = amt128 * num;
231  auto r = m / den;
232  if (m % den)
233  {
234  if (!neg && roundUp)
235  r += 1;
236  if (neg && !roundUp)
237  r -= 1;
238  }
239  if (r > std::numeric_limits<MPTAmount::mpt_type>::max())
240  Throw<std::overflow_error>("XRP mulRatio overflow");
241  return MPTAmount(r.convert_to<MPTAmount::mpt_type>());
242 }
