68  : env_(env)

71  , close_(arg.close)

73  if (arg.fund)

92 void
93 MPTTester::create(const MPTCreate& arg)
94 {
95  if (issuanceKey_)
96  Throw<std::runtime_error>("MPT can't be reused");
97  mpt_ = std::make_pair(env_.seq(issuer_), issuer_.id());
98  id_ = getMptID(issuer_.id(), mpt_->first);
99  issuanceKey_ = keylet::mptIssuance(*id_).key;
100  Json::Value jv;
101  jv[sfAccount.jsonName] = issuer_.human();
102  jv[sfTransactionType.jsonName] = jss::MPTokenIssuanceCreate;
103  if (arg.assetScale)
104  jv[sfAssetScale.jsonName] = *arg.assetScale;
105  if (arg.transferFee)
106  jv[sfTransferFee.jsonName] = *arg.transferFee;
107  if (arg.metadata)
108  jv[sfMPTokenMetadata.jsonName] = strHex(*arg.metadata);
109 
110  // convert maxAmt to hex string, since json doesn't accept 64-bit int
111  if (arg.maxAmt)
112  jv[sfMaximumAmount.jsonName] = strHex(uint64ToByteArray(*arg.maxAmt));
113  if (submit(arg, jv) != tesSUCCESS)
114  {
115  // Verify issuance doesn't exist
116  env_.require(requireAny([&]() -> bool {
117  return env_.le(keylet::mptIssuance(*id_)) == nullptr;
118  }));
119 
120  id_.reset();
121  issuanceKey_.reset();
122  mpt_.reset();
123  }
124  else if (arg.flags)
125  env_.require(mptflags(*this, *arg.flags));
126 }

157 void
158 MPTTester::authorize(MPTAuthorize const& arg)
159 {
160  Json::Value jv;
161  if (arg.account)
162  jv[sfAccount.jsonName] = arg.account->human();
163  else
164  jv[sfAccount.jsonName] = issuer_.human();
165  jv[sfTransactionType.jsonName] = jss::MPTokenAuthorize;
166  if (arg.id)
167  jv[sfMPTokenIssuanceID.jsonName] = to_string(*arg.id);
168  else
169  {
170  assert(id_);
171  jv[sfMPTokenIssuanceID.jsonName] = to_string(*id_);
172  }
173  if (arg.holder)
174  jv[sfMPTokenHolder.jsonName] = arg.holder->human();
175  if (auto const result = submit(arg, jv); result == tesSUCCESS)
176  {
177  // Issuer authorizes
178  if (arg.account == nullptr || *arg.account == issuer_)
179  {
180  auto const flags = getFlags(arg.holder);
181  // issuer un-authorizes the holder
182  if (arg.flags.value_or(0) == tfMPTUnauthorize)
183  env_.require(mptflags(*this, flags, arg.holder));
184  // issuer authorizes the holder
185  else
186  env_.require(
187  mptflags(*this, flags | lsfMPTAuthorized, arg.holder));
188  }
189  // Holder authorizes
190  else if (arg.flags.value_or(0) == 0)
191  {
192  auto const flags = getFlags(arg.account);
193  // holder creates a token
194  env_.require(mptflags(*this, flags, arg.account));
195  env_.require(mptpay(*this, *arg.account, 0));
196  }
197  }
198  else if (
199  arg.account != nullptr && *arg.account != issuer_ &&
200  arg.flags.value_or(0) == 0)
201  {
202  if (result == tecMPTOKEN_EXISTS)
203  {
204  // Verify that MPToken already exists
205  env_.require(requireAny([&]() -> bool {
206  return env_.le(keylet::mptoken(
207  *issuanceKey_, arg.account->id())) != nullptr;
208  }));
209  }
210  else
211  {
212  // Verify MPToken doesn't exist if holder failed authorizing(unless
213  // it already exists)
214  env_.require(requireAny([&]() -> bool {
215  return env_.le(keylet::mptoken(
216  *issuanceKey_, arg.account->id())) == nullptr;
217  }));
218  }
219  }
220 }

222 void
223 MPTTester::set(MPTSet const& arg)
224 {
225  Json::Value jv;
226  if (arg.account)
227  jv[sfAccount.jsonName] = arg.account->human();
228  else
229  jv[sfAccount.jsonName] = issuer_.human();
230  jv[sfTransactionType.jsonName] = jss::MPTokenIssuanceSet;
231  if (arg.id)
232  jv[sfMPTokenIssuanceID.jsonName] = to_string(*arg.id);
233  else
234  {
235  assert(id_);
236  jv[sfMPTokenIssuanceID.jsonName] = to_string(*id_);
237  }
238  if (arg.holder)
239  jv[sfMPTokenHolder.jsonName] = arg.holder->human();
240  if (submit(arg, jv) == tesSUCCESS && arg.flags.value_or(0))
241  {
242  auto require = [&](AccountP holder, bool unchanged) {
243  auto flags = getFlags(holder);
244  if (!unchanged)
245  {
246  if (*arg.flags & tfMPTLock)
247  flags |= lsfMPTLocked;
248  else if (*arg.flags & tfMPTUnlock)
249  flags &= ~lsfMPTLocked;
250  else
251  assert(0);
252  }
253  env_.require(mptflags(*this, flags, holder));
254  };
255  if (arg.account)
256  require(nullptr, arg.holder != nullptr);
257  if (arg.holder)
258  require(arg.holder, false);
259  }
260 }

302 void
303 MPTTester::pay(
304  Account const& src,
305  Account const& dest,
306  std::uint64_t amount,
307  std::optional<TER> err)
308 {
309  assert(mpt_);
310  auto const srcAmt = getAmount(src);
311  auto const destAmt = getAmount(dest);
312  auto const outstnAmt = getAmount(issuer_);
313  if (err)
314  env_(jtx::pay(src, dest, mpt(amount)), ter(*err));
315  else
316  env_(jtx::pay(src, dest, mpt(amount)));
317  if (env_.ter() != tesSUCCESS)
318  amount = 0;
319  if (close_)
320  env_.close();
321  if (src == issuer_)
322  {
323  env_.require(mptpay(*this, src, srcAmt + amount));
324  env_.require(mptpay(*this, dest, destAmt + amount));
325  }
326  else if (dest == issuer_)
327  {
328  env_.require(mptpay(*this, src, srcAmt - amount));
329  env_.require(mptpay(*this, dest, destAmt - amount));
330  }
331  else
332  {
333  // TODO MPT
334 #if 0
335  STAmount const saAmount = {Issue{*mpt_}, amount};
336  STAmount const saActual =
337  multiply(saAmount, transferRateMPT(*env_.current(), *mpt_));
338  // Sender pays the transfer fee if any
339  env_.require(mptpay(*this, src, srcAmt - saActual.mpt().mpt()));
340  env_.require(mptpay(*this, dest, destAmt + amount));
341  // Outstanding amount is reduced by the transfer fee if any
342  env_.require(mptpay(
343  *this, issuer_, outstnAmt - (saActual - saAmount).mpt().mpt()));
344 #endif
345  }
346 }
